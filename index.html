Igo

Interfaces and fakes for standard Golang packages

What is Igo?

Igo is a Golang package that provides an interface over standard, built-in Golang packages that allows for easy faking. Fakes are generated using counterfeiter and provided within Igo's packages.


type Thing struct{
  ...
  os ios.OS
}

func NewThing() *Thing {
  return &Thing{
    exec: ios.New()
  }
}

func (thing *Thing) Do() {
  thing.exec.Remove("/some/file")
}

...

It("doesn't return an error", func() {
  err := thing.Do()
  Expect(err).NotTo(HaveOccurred())
})

It("returns an error", func() {
  fakeOS := ios.NewFake()
  fakeOS.RemoveReturns(errors.New("I failed!"))
  thing.exec = fakeOS

  err := thing.Do()
  Expect(err).To(HaveOccurred())
})
